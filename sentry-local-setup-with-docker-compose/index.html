Sentry local setup with docker compose

Docker compose

version: "3.9"

services:
  postgres:
    image: postgres:14
    container_name: sentry_postgres
    restart: always
    environment:
      POSTGRES_USER: ${SENTRY_DB_USER}
      POSTGRES_PASSWORD: ${SENTRY_DB_PASSWORD}
      POSTGRES_DB: ${SENTRY_DB_NAME}
      POSTGRES_HOST_AUTH_METHOD: trust
    volumes:
      - sentry-postgres:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${SENTRY_DB_USER:-postgres}"]

  redis:
    image: redis:7
    container_name: sentry_redis
    restart: always
    volumes:
      - sentry-redis:/data
    healthcheck:
      test: ["CMD-SHELL", "redis-cli ping | grep PONG"]

  zookeeper:
    image: confluentinc/cp-zookeeper:7.6.6
    container_name: sentry_zookeeper
    restart: always
    environment:
      ZOOKEEPER_CLIENT_PORT: 2181
      ZOOKEEPER_TICK_TIME: 2000
    healthcheck:
      test: ["CMD-SHELL", "nc -z localhost 2181 || exit 1"]

  kafka:
    image: confluentinc/cp-kafka:7.6.6
    container_name: sentry_kafka
    restart: always
    depends_on:
      - zookeeper
    environment:
      KAFKA_ZOOKEEPER_CONNECT: zookeeper:2181
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://kafka:9092
      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      KAFKA_TRANSACTION_STATE_LOG_MIN_ISR: 1
      KAFKA_TRANSACTION_STATE_LOG_REPLICATION_FACTOR: 1
    volumes:
      - sentry-kafka:/var/lib/kafka/data
    healthcheck:
      test: ["CMD-SHELL", "nc -z localhost 9092 || exit 1"]

  clickhouse:
    image: clickhouse/clickhouse-server:23.8.6.39
    container_name: sentry_clickhouse
    restart: always
    ulimits:
      nofile:
        soft: 262144
        hard: 262144
    volumes:
      - sentry-clickhouse:/var/lib/clickhouse
    healthcheck:
      test: ["CMD-SHELL", "curl -sS http://localhost:8123/ || exit 1"]

  snuba-api:
    image: getsentry/snuba:latest
    container_name: sentry_snuba_api
    restart: always
    depends_on:
      - clickhouse
      - kafka
      - redis
    environment:
      SNUBA_SETTINGS: snuba.settings
      CLICKHOUSE_HOST: clickhouse
      KAFKA_HOST: kafka
      REDIS_HOST: redis
    command: ["snuba", "api"]
    ports:
      - "1218:1218"

  snuba-consumer:
    image: getsentry/snuba:latest
    container_name: sentry_snuba_consumer
    restart: always
    depends_on:
      - snuba-api
      - kafka
      - clickhouse
      - redis
    environment:
      SNUBA_SETTINGS: snuba.settings
      CLICKHOUSE_HOST: clickhouse
      KAFKA_HOST: kafka
      REDIS_HOST: redis
    command: ["snuba", "consumer"]

  symbolicator:
    image: getsentry/symbolicator:latest
    container_name: sentry_symbolicator
    restart: always
    ports:
      - "3021:3021"
    volumes:
      - sentry-symbolicator:/var/lib/symbolicator

  relay:
    image: getsentry/relay:latest
    container_name: sentry_relay
    restart: always
    env_file:
      - .env
    ports:
      - "3000:3000"

  sentry:
    image: getsentry/sentry:latest
    container_name: sentry_app
    depends_on:
      - postgres
      - redis
      - kafka
      - clickhouse
      - snuba-api
    env_file:
      - .env
    ports:
      - "9000:9000"
    volumes:
      - sentry-data:/var/lib/sentry/files
    command: >
      sh -c "sentry upgrade --noinput && \
             sentry createuser --no-input --superuser --email $SENTRY_ADMIN_EMAIL --password $SENTRY_ADMIN_PASSWORD || true && \
             sentry run web"

  worker:
    image: getsentry/sentry:latest
    container_name: sentry_worker
    depends_on:
      - sentry
      - kafka
      - clickhouse
    env_file:
      - .env
    command: "sentry run worker"

  cron:
    image: getsentry/sentry:latest
    container_name: sentry_cron
    depends_on:
      - sentry
    env_file:
      - .env
    command: "sentry run cron"

volumes:
  sentry-postgres:
  sentry-redis:
  sentry-kafka:
  sentry-clickhouse:
  sentry-data:
  sentry-symbolicator:

networks:
  default:
    name: sentry_network
